21   class ThreeLayerConvNet(object):
22     """ 
23     A three-layer convolutional network with the following architecture: 
24      
25     conv - relu - 2x2 max pool - affine - relu - affine - softmax 
26      
27     The network operates on minibatches of data that have shape (N, C, H, W) 
28     consisting of N images, each with height H and width W and with C input 
29     channels. 
30     """
31     
32     def __init__(self, input_dim=(3, 32, 32), num_filters=32, filter_size=7,
33                  hidden_dim=100, num_classes=10, weight_scale=1e-3, reg=0.0,
34                  dtype=np.float32, use_batchnorm=False):
35       """ 
36       Initialize a new network. 
37        
38       Inputs: 
39       - input_dim: Tuple (C, H, W) giving size of input data 
40       - num_filters: Number of filters to use in the convolutional layer 
41       - filter_size: Size of filters to use in the convolutional layer 
42       - hidden_dim: Number of units to use in the fully-connected hidden layer 
43       - num_classes: Number of scores to produce from the final affine layer. 
44       - weight_scale: Scalar giving standard deviation for random initialization 
45         of weights. 
46       - reg: Scalar giving L2 regularization strength 
47       - dtype: numpy datatype to use for computation. 
48       """
49       self.use_batchnorm = use_batchnorm
50       self.params = {}
51       self.reg = reg
52       self.dtype = dtype
53   
54       
55       # ================================================================ #
56       # YOUR CODE HERE:
57       #   Initialize the weights and biases of a three layer CNN. To initialize:
58       #     - the biases should be initialized to zeros.
59       #     - the weights should be initialized to a matrix with entries
60       #         drawn from a Gaussian distribution with zero mean and 
61       #         standard deviation given by weight_scale.
62       # ================================================================ #
63   
64       C, H, W = input_dim
65       stride, pad = 1, (filter_size - 1) / 2
66       H_out = int((H + 2 * pad - filter_size) / stride + 1)
67       W_out = int((W + 2 * pad - filter_size) / stride + 1)
68       pool_out_h = int((H_out - 2) / 2 + 1)
69       pool_out_w = int((W_out - 2) / 2 + 1)
70   
71       self.params['W1'] = np.random.normal(0, weight_scale, (num_filters, C, filter_size, filter_size))
72       self.params['b1'] = np.zeros(num_filters)
73       self.params['W2'] = np.random.normal(0, weight_scale, (num_filters * pool_out_h * pool_out_w, hidden_dim))
74       self.params['b2'] = np.zeros(hidden_dim)
75       self.params['W3'] = np.random.normal(0, weight_scale, (hidden_dim, num_classes))
76       self.params['b3'] = np.zeros(num_classes)
77   
78       # ================================================================ #
79       # END YOUR CODE HERE
80       # ================================================================ #
81   
82       for k, v in self.params.items():
83         self.params[k] = v.astype(dtype)
84        
85    
86     def loss(self, X, y=None):
87       """ 
88       Evaluate loss and gradient for the three-layer convolutional network. 
89        
90       Input / output: Same API as TwoLayerNet in fc_net.py. 
91       """
92       W1, b1 = self.params['W1'], self.params['b1']
93       W2, b2 = self.params['W2'], self.params['b2']
94       W3, b3 = self.params['W3'], self.params['b3']
95       
96       # pass conv_param to the forward pass for the convolutional layer
97       filter_size = W1.shape[2]
98       conv_param = {'stride': 1, 'pad': (filter_size - 1) / 2}
99   
100      # pass pool_param to the forward pass for the max-pooling layer
101      pool_param = {'pool_height': 2, 'pool_width': 2, 'stride': 2}
102  
103      scores = None
104      
105      # ================================================================ #
106      # YOUR CODE HERE:
107      #   Implement the forward pass of the three layer CNN.  Store the output
108      #   scores as the variable "scores".
109      # ================================================================ #
110      
111      out, cache1 = conv_relu_pool_forward(X, W1, b1, conv_param, pool_param)
112      out, cache2 = affine_relu_forward(out, W2, b2)
113      scores, cache3 = affine_forward(out, W3, b3)
114  
115      # ================================================================ #
116      # END YOUR CODE HERE
117      # ================================================================ #
118  
119      if y is None:
120        return scores
121      
122      loss, grads = 0, {}
123      # ================================================================ #
124      # YOUR CODE HERE:
125      #   Implement the backward pass of the three layer CNN.  Store the grads
126      #   in the grads dictionary, exactly as before (i.e., the gradient of 
127      #   self.params[k] will be grads[k]).  Store the loss as "loss", and
128      #   don't forget to add regularization on ALL weight matrices.
129      # ================================================================ #
130  
131      loss, dout = softmax_loss(scores, y)
132      loss += 0.5 * self.reg * (np.sum(W1 ** 2) + np.sum(W2 ** 2) + np.sum(W3 ** 2))
133      dout, dW3, db3 = affine_backward(dout, cache3)
134      dout, dW2, db2 = affine_relu_backward(dout, cache2)
135      dout, dW1, db1 = conv_relu_pool_backward(dout, cache1)
136      grads['W1'] = dW1 + self.reg * W1
137      grads['b1'] = db1
138      grads['W2'] = dW2 + self.reg * W2
139      grads['b2'] = db2
140      grads['W3'] = dW3 + self.reg * W3
141      grads['b3'] = db3
142  
143      # ================================================================ #
144      # END YOUR CODE HERE
145      # ================================================================ #
146  
147      return loss, grads
