14   def affine_relu_forward(x, w, b):
15       """ 
16     Convenience layer that performs an affine transform followed by a ReLU 
17    
18     Inputs: 
19     - x: Input to the affine layer 
20     - w, b: Weights for the affine layer 
21    
22     Returns a tuple of: 
23     - out: Output from the ReLU 
24     - cache: Object to give to the backward pass 
25     """
26       a, fc_cache = affine_forward(x, w, b)
27       out, relu_cache = relu_forward(a)
28       cache = (fc_cache, relu_cache)
29       return out, cache

32   def affine_relu_backward(dout, cache):
33       """ 
34     Backward pass for the affine-relu convenience layer 
35     """
36       fc_cache, relu_cache = cache
37       da = relu_backward(dout, relu_cache)
38       dx, dw, db = affine_backward(da, fc_cache)
39       return dx, dw, db

42   def affine_bn_relu_forward(x, w, b, gamma, beta, bn_param):
43       """ 
44       Inputs: 
45       - x: Input to the affine layer 
46       - w, b: Weights for the affine layer 
47       - gamma, beta: Scale and shift parameters for the batch normalization layer 
48       - bn_param: Dictionary with parameters for the batch normalization layer 
49    
50       Returns a tuple of: 
51       - out: Output from the ReLU 
52       - cache: Object to give to the backward pass 
53       """
54       out1, cache1 = affine_forward(x, w, b)
55       bn_out, bn_cache = batchnorm_forward(out1, gamma, beta, bn_param)
56       out, relu_cache = relu_forward(bn_out)
57       cache = (cache1, bn_cache, relu_cache)
58       return out, cache

61   def affine_bn_relu_backward(dout, cache):
62       """ 
63       Inputs: 
64       - dout: Upstream derivative 
65       - cache: Tuple of caches from the forward pass 
66    
67       Returns a tuple of: 
68       - dx: Gradient with respect to x 
69       - dw: Gradient with respect to w 
70       - db: Gradient with respect to b 
71       - dgamma: Gradient with respect to gamma 
72       - dbeta: Gradient with respect to beta 
73       """
74       cache1, bn_cache, relu_cache = cache
75       drelu = relu_backward(dout, relu_cache)
76       dbn, dgamma, dbeta = batchnorm_backward(drelu, bn_cache)
77       dx, dw, db = affine_backward(dbn, cache1)
78       return dx, dw, db, dgamma, dbeta
