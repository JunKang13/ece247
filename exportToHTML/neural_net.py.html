5    class TwoLayerNet(object):
6        """ 
7      A two-layer fully-connected neural network. The net has an input dimension of 
8      D, a hidden layer dimension of H, and performs classification over C classes. 
9      We train the network with a softmax loss function and L2 regularization on the 
10     weight matrices. The network uses a ReLU nonlinearity after the first fully 
11     connected layer. 
12    
13     In other words, the network has the following architecture: 
14    
15     input - fully connected layer - ReLU - fully connected layer - softmax 
16    
17     The outputs of the second fully-connected layer are the scores for each class. 
18     """
19   
20       def __init__(self, input_size, hidden_size, output_size, std=1e-4):
21           """ 
22       Initialize the model. Weights are initialized to small random values and 
23       biases are initialized to zero. Weights and biases are stored in the 
24       variable self.params, which is a dictionary with the following keys: 
25    
26       W1: First layer weights; has shape (H, D) 
27       b1: First layer biases; has shape (H,) 
28       W2: Second layer weights; has shape (C, H) 
29       b2: Second layer biases; has shape (C,) 
30    
31       Inputs: 
32       - input_size: The dimension D of the input data. 
33       - hidden_size: The number of neurons H in the hidden layer. 
34       - output_size: The number of classes C. 
35       """
36           self.params = {}
37           self.params['W1'] = std * np.random.randn(hidden_size, input_size)
38           self.params['b1'] = np.zeros(hidden_size)
39           self.params['W2'] = std * np.random.randn(output_size, hidden_size)
40           self.params['b2'] = np.zeros(output_size)
41   
42       def loss(self, X, y=None, reg=0.0):
43           """ 
44       Compute the loss and gradients for a two layer fully connected neural 
45       network. 
46    
47       Inputs: 
48       - X: Input data of shape (N, D). Each X[i] is a training sample. 
49       - y: Vector of training labels. y[i] is the label for X[i], and each y[i] is 
50         an integer in the range 0 <= y[i] < C. This parameter is optional; if it 
51         is not passed then we only return scores, and if it is passed then we 
52         instead return the loss and gradients. 
53       - reg: Regularization strength. 
54    
55       Returns: 
56       If y is None, return a matrix scores of shape (N, C) where scores[i, c] is 
57       the score for class c on input X[i]. 
58    
59       If y is not None, instead return a tuple of: 
60       - loss: Loss (data loss and regularization loss) for this batch of training 
61         samples. 
62       - grads: Dictionary mapping parameter names to gradients of those parameters 
63         with respect to the loss function; has the same keys as self.params. 
64       """
65           # Unpack variables from the params dictionary
66           W1, b1 = self.params['W1'], self.params['b1']
67           W2, b2 = self.params['W2'], self.params['b2']
68           N, D = X.shape
69           # Compute the forward pass
70           scores = None
71   
72           # ================================================================ #
73           # YOUR CODE HERE:
74           #   Calculate the output scores of the neural network.  The result
75           #   should be (N, C). As stated in the description for this class,
76           #   there should not be a ReLU layer after the second FC layer.
77           #   The output of the second FC layer is the output scores. Do not
78           #   use a for loop in your implementation.
79           # ================================================================ #
80   
81           scores = np.dot(W1, X.T) + b1.reshape(-1, 1)
82           scores = np.maximum(0, scores)
83           scores = np.dot(W2, scores) + b2.reshape(-1, 1)
84           scores = scores.T
85   
86           # ================================================================ #
87           # END YOUR CODE HERE
88           # ================================================================ #
89   
90           # If the targets are not given then jump out, we're done
91           if y is None:
92               return scores
93   
94           # Compute the loss
95           loss = None
96   
97           # ================================================================ #
98           # YOUR CODE HERE:
99           #   Calculate the loss of the neural network.  This includes the
100          #   softmax loss and the L2 regularization for W1 and W2. Store the
101          #   total loss in teh variable loss.  Multiply the regularization
102          #   loss by 0.5 (in addition to the factor reg).
103          # ================================================================ #
104  
105          # scores is num_examples by num_classes
106          softmax_probs = np.exp(scores) / np.sum(np.exp(scores), axis=1, keepdims=True)
107          softmax_loss = -np.log(softmax_probs[np.arange(N), y]).sum() / N
108          reg_loss = 0.5 * reg * (np.sum(W1 * W1) + np.sum(W2 * W2))
109          loss = softmax_loss + reg_loss
110  
111          # ================================================================ #
112          # END YOUR CODE HERE
113          # ================================================================ #
114  
115          grads = {}
116  
117          # ================================================================ #
118          # YOUR CODE HERE:
119          #   Implement the backward pass.  Compute the derivatives of the
120          #   weights and the biases.  Store the results in the grads
121          #   dictionary.  e.g., grads['W1'] should store the gradient for
122          #   W1, and be of the same size as W1.
123          # ================================================================ #
124  
125          softmax_probs[np.arange(X.shape[0]), y] -= 1
126          softmax_grad = softmax_probs.T / N
127          grads['b2'] = softmax_grad.sum(axis=1)
128          grads['W2'] = softmax_grad @ np.maximum(0, np.dot(W1, X.T) + b1.reshape(-1, 1)).T
129  
130          relu_grad = (np.dot(W2.T, softmax_grad) * (np.dot(W1, X.T) + b1.reshape(-1, 1) > 0))
131          grads['b1'] = relu_grad.sum(axis=1)
132          grads['W1'] = relu_grad @ X
133  
134          grads['W1'] += reg * W1
135          grads['W2'] += reg * W2
136  
137          # ================================================================ #
138          # END YOUR CODE HERE
139          # ================================================================ #
140  
141          return loss, grads
142  
143      def train(self, X, y, X_val, y_val,
144                learning_rate=1e-3, learning_rate_decay=0.95,
145                reg=1e-5, num_iters=100,
146                batch_size=200, verbose=False):
147          """ 
148      Train this neural network using stochastic gradient descent. 
149   
150      Inputs: 
151      - X: A numpy array of shape (N, D) giving training data. 
152      - y: A numpy array f shape (N,) giving training labels; y[i] = c means that 
153        X[i] has label c, where 0 <= c < C. 
154      - X_val: A numpy array of shape (N_val, D) giving validation data. 
155      - y_val: A numpy array of shape (N_val,) giving validation labels. 
156      - learning_rate: Scalar giving learning rate for optimization. 
157      - learning_rate_decay: Scalar giving factor used to decay the learning rate 
158        after each epoch. 
159      - reg: Scalar giving regularization strength. 
160      - num_iters: Number of steps to take when optimizing. 
161      - batch_size: Number of training examples to use per step. 
162      - verbose: boolean; if true print progress during optimization. 
163      """
164          num_train = X.shape[0]
165          iterations_per_epoch = max(num_train / batch_size, 1)
166  
167          # Use SGD to optimize the parameters in self.model
168          loss_history = []
169          train_acc_history = []
170          val_acc_history = []
171  
172          for it in np.arange(num_iters):
173              X_batch = None
174              y_batch = None
175  
176              # ================================================================ #
177              # YOUR CODE HERE:
178              #   Create a minibatch by sampling batch_size samples randomly.
179              # ================================================================ #
180              np.random.seed(it)
181              rand_indices = np.random.choice(num_train, batch_size, replace=True)
182              X_batch = X[rand_indices]
183              y_batch = y[rand_indices]
184  
185              # ================================================================ #
186              # END YOUR CODE HERE
187              # ================================================================ #
188  
189              # Compute loss and gradients using the current minibatch
190              loss, grads = self.loss(X_batch, y=y_batch, reg=reg)
191              loss_history.append(loss)
192  
193              # ================================================================ #
194              # YOUR CODE HERE:
195              #   Perform a gradient descent step using the minibatch to update
196              #   all parameters (i.e., W1, W2, b1, and b2).
197              # ================================================================ #
198  
199              self.params['W1'] -= learning_rate * grads['W1']
200              self.params['b1'] -= learning_rate * grads['b1']
201              self.params['W2'] -= learning_rate * grads['W2']
202              self.params['b2'] -= learning_rate * grads['b2']
203  
204              # ================================================================ #
205              # END YOUR CODE HERE
206              # ================================================================ #
207  
208              if verbose and it % 100 == 0:
209                  print('iteration {} / {}: loss {}'.format(it, num_iters, loss))
210  
211              # Every epoch, check train and val accuracy and decay learning rate.
212              if it % iterations_per_epoch == 0:
213                  # Check accuracy
214                  train_acc = (self.predict(X_batch) == y_batch).mean()
215                  val_acc = (self.predict(X_val) == y_val).mean()
216                  train_acc_history.append(train_acc)
217                  val_acc_history.append(val_acc)
218  
219                  # Decay learning rate
220                  learning_rate *= learning_rate_decay
221  
222          return {
223              'loss_history': loss_history,
224              'train_acc_history': train_acc_history,
225              'val_acc_history': val_acc_history,
226          }

228      def predict(self, X):
229          """ 
230      Use the trained weights of this two-layer network to predict labels for 
231      data points. For each data point we predict scores for each of the C 
232      classes, and assign each data point to the class with the highest score. 
233   
234      Inputs: 
235      - X: A numpy array of shape (N, D) giving N D-dimensional data points to 
236        classify. 
237   
238      Returns: 
239      - y_pred: A numpy array of shape (N,) giving predicted labels for each of 
240        the elements of X. For all i, y_pred[i] = c means that X[i] is predicted 
241        to have class c, where 0 <= c < C. 
242      """
243          y_pred = None
244  
245          # ================================================================ #
246          # YOUR CODE HERE:
247          #   Predict the class given the input data.
248          # ================================================================ #
249          scores = self.loss(X)
250          y_pred = np.argmax(scores, axis=1)
251  
252          # ================================================================ #
253          # END YOUR CODE HERE
254          # ================================================================ #
255  
256          return y_pred
